subroutine mat_formz(x,y)

! ident_18="@(#) M_matrix mat_formz system dependent routine to print with z format"

doubleprecision,intent(in) :: x,y

character(len=36)          :: mline

   if (y .ne. 0.0d0) then
      write(mline,'(2z18)') x,y
   else
      write(mline,'(z18)') x
   endif

   call journal(mline)

end subroutine mat_formz
